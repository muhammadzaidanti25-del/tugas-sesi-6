# SOAL 2: Menghitung total harga tiket
print("\n" + "=" * 50)
print("PROGRAM TIKET TAMAN HIBURAN")
print("=" * 50)

tiket1 = int(input("Isikan Umur Tiket 1: "))
tiket2 = int(input("Isikan Umur Tiket 2: "))
tiket3 = int(input("Isikan Umur Tiket 3: "))

total_harga = 0

# Fungsi untuk menentukan harga berdasarkan umur
def hitung_harga(umur):
    if umur >= 0 and umur <= 12:
        return 10000
    elif umur >= 13 and umur <= 17:
        return 15000
    elif umur >= 18 and umur <= 59:
        return 25000
    elif umur >= 60:
        return 0
    else:
        print(f"Umur {umur} tidak valid!")
        return 0

# Hitung harga masing-masing tiket
harga1 = hitung_harga(tiket1)
harga2 = hitung_harga(tiket2)
harga3 = hitung_harga(tiket3)

total_harga = harga1 + harga2 + harga3

print("\n" + "-" * 50)
print("RINCIAN HARGA TIKET")
print("-" * 50)
print(f"Tiket 1 (Umur {tiket1} tahun): Rp {harga1:,}")
print(f"Tiket 2 (Umur {tiket2} tahun): Rp {harga2:,}")
print(f"Tiket 3 (Umur {tiket3} tahun): Rp {harga3:,}")
print("-" * 50)
print(f"TOTAL HARGA: Rp {total_harga:,}")
print("=" * 50)

# SOAL 3: Diskon restoran berdasarkan usia
print("\n" + "=" * 50)
print("PROGRAM DISKON RESTORAN")
print("=" * 50)

usia = int(input("Masukkan usia pelanggan: "))
harga_makanan = float(input("Masukkan total harga makanan: Rp "))

if usia >= 0 and usia <= 12:
    diskon = 0.5  # 50%
    keterangan = "Anak-anak (diskon 50%)"
elif usia >= 60:
    diskon = 0.3  # 30%
    keterangan = "Lansia (diskon 30%)"
else:
    diskon = 0
    keterangan = "Tidak ada diskon"

harga_setelah_diskon = harga_makanan * (1 - diskon)

print("\n" + "-" * 50)
print("HASIL PERHITUNGAN")
print("-" * 50)
print(f"Usia pelanggan: {usia} tahun")
print(f"Kategori: {keterangan}")
print(f"Harga awal: Rp {harga_makanan:,.0f}")
if diskon > 0:
    print(f"Diskon: {diskon*100:.0f}%")
    print(f"Harga setelah diskon: Rp {harga_setelah_diskon:,.0f}")
else:
    print(f"Total bayar: Rp {harga_setelah_diskon:,.0f}")
print("=" * 50)

# SOAL 10: Nomor Punggung Pemain PERSEGI FC
print("\n" + "=" * 50)
print("PENENTUAN POSISI PEMAIN PERSEGI FC")
print("Berdasarkan Nomor Punggung")
print("=" * 50)

nomor = int(input("Masukkan nomor punggung pemain: "))

print(f"\nNomor punggung: {nomor}")
print("-" * 50)
print("POSISI YANG DAPAT DITEMPATI:")
print("-" * 50)

posisi = []

# Cek genap/ganjil
if nomor % 2 == 0:
    # Genap: target attacker
    posisi.append("Target Attacker")
    print("✓ Target Attacker (nomor genap)")
    
    # Genap antara 50-100: berhak jadi kapten
    if nomor >= 50 and nomor <= 100:
        posisi.append("Berhak dipilih menjadi Captain Team")
        print("✓ Berhak dipilih menjadi Captain Team (genap 50-100)")
else:
    # Ganjil: defender
    posisi.append("Defender")
    print("✓ Defender (nomor ganjil)")
    
    # Ganjil > 90: playmaker
    if nomor > 90:
        posisi.append("Playmaker")
        print("✓ Playmaker (ganjil > 90)")
    
    # Ganjil kelipatan 3 dan 5 (kelipatan 15): keeper
    if nomor % 3 == 0 and nomor % 5 == 0:
        posisi.append("Keeper")
        print("✓ Keeper (ganjil kelipatan 3 dan 5)")

print("-" * 50)
if len(posisi) > 0:
    print(f"Pemain bernomor {nomor} dapat berperan sebagai:")
    for i, p in enumerate(posisi, 1):
        print(f"  {i}. {p}")
else:
    print("Tidak ada posisi khusus untuk nomor ini.")
print("=" * 50)
